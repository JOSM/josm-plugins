075                /**
076                 * The APIv3 returns a Link header for each request. It contains a URL for requesting more results.
077                 * If you supply the value of the Link header, this method returns the next URL,
078                 * if such a URL is defined in the header.
079                 * @param value the value of the HTTP-header with key "Link"
080                 * @return the {@link URL} for the next result page, or <code>null</code> if no such URL could be found
081                 */
082                public static URL parseNextFromLinkHeaderValue(String value) {
083                        if (value != null) {
084                                // Iterate over the different entries of the Link header
085                                for (final String link : value.split(",", Integer.MAX_VALUE)) {
086                                        boolean isNext = false;
087                                        URL url = null;
088                                        // Iterate over the parts of each entry (typically it's one `rel="‹linkType›"` and one like `<https://URL>`)
089                                        for (String linkPart : link.split(";", Integer.MAX_VALUE)) {
090                                                linkPart = linkPart.trim();
091                                                isNext |= linkPart.matches("rel\\s*=\\s*\"next\"");
092                                                if (linkPart.length() >= 1 && linkPart.charAt(0) == '<' && linkPart.endsWith(">")) {
093                                                        try {
094                                                                url = new URL(linkPart.substring(1, linkPart.length() - 1));
095                                                        } catch (final MalformedURLException e) {
096                                                                Logging.log(Logging.LEVEL_WARN, "Mapillary API v3 returns a malformed URL in the Link header.", e);
097                                                        }
098                                                }
099                                        }
100                                        // If both a URL and the rel=next attribute are present, return the URL. Otherwise null is returned
101                                        if (url != null && isNext) {
102                                                return url;
103                                        }
104                                }
105                        }
106                        return null;
107                }

292        /**
293         * Builds a query string from it's parts that are supplied as a {@link Map}
294         * @param parts the parts of the query string
295         * @return the constructed query string (including a leading ?)
296         */
297        static String queryString(Map<String, String> parts) {
298                final StringBuilder ret = new StringBuilder("?client_id=").append(StreetsideURL.CLIENT_ID);
299                if (parts != null) {
300                        for (final Entry<String, String> entry : parts.entrySet()) {
301                                try {
302                                        ret.append('&')
303                                        .append(URLEncoder.encode(entry.getKey(), StandardCharsets.UTF_8.name()))
304                                        .append('=')
305                                        .append(URLEncoder.encode(entry.getValue(), StandardCharsets.UTF_8.name()));
306                                } catch (final UnsupportedEncodingException e) {
307                                        Logging.error(e); // This should not happen, as the encoding is hard-coded
308                                }
309                        }
310                }
311                return ret.toString();
312        }

314        static String queryStreetsideBoundsString(Map<String, String> parts) {
315                final StringBuilder ret = new StringBuilder("?n=");
316                if (parts != null) {
317                        final List<String> bbox = new ArrayList<>(Arrays.asList(parts.get("bbox").split(",")));
318                        try {
319                                ret.append(URLEncoder.encode(bbox.get(StreetsideURL.OSM_BBOX_NORTH), StandardCharsets.UTF_8.name()))
320                                .append("&s=")
321                                .append(URLEncoder.encode(bbox.get(StreetsideURL.OSM_BBOX_SOUTH), StandardCharsets.UTF_8.name()))
322                                .append("&e=")
323                                .append(URLEncoder.encode(bbox.get(StreetsideURL.OSM_BBOXEAST), StandardCharsets.UTF_8.name()))
324                                .append("&w=")
325                                .append(URLEncoder.encode(bbox.get(StreetsideURL.OSM_BBOX_WEST), StandardCharsets.UTF_8.name()))
326                                .append("&c=1000")
327                                .append("&appkey=")
328                                .append(StreetsideURL.BING_MAPS_KEY);
329                        } catch (final UnsupportedEncodingException e) {
330                                Logging.error(e); // This should not happen, as the encoding is hard-coded
331                        }
332                }
333                return ret.toString();
334        }

348        /**
349         * Converts a {@link String} into a {@link URL} without throwing a {@link MalformedURLException}.
350         * Instead such an exception will lead to an {@link Logging#error(Throwable)}.
351         * So you should be very confident that your URL is well-formed when calling this method.
352         * @param strings the Strings describing the URL
353         * @return the URL that is constructed from the given string
354         */
355        static URL string2URL(String... strings) {
356                final StringBuilder builder = new StringBuilder();
357                for (int i = 0; strings != null && i < strings.length; i++) {
358                        builder.append(strings[i]);
359                }
360                try {
361                        return new URL(builder.toString());
362                } catch (final MalformedURLException e) {
363                        Logging.log(Logging.LEVEL_ERROR, String.format(
364                                        "The class '%s' produces malformed URLs like '%s'!",
365                                        StreetsideURL.class.getName(),
366                                        builder
367                                        ), e);
368                        return null;
369                }
370        }
