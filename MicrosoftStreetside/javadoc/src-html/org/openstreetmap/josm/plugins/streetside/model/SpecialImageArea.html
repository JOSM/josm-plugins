002package org.openstreetmap.josm.plugins.streetside.model;
003
004import java.awt.geom.Path2D;
005
006public class SpecialImageArea extends KeyIndexedObject {
007  private final String imageKey;
008  private final Path2D shape;
009
010  protected SpecialImageArea(final Path2D shape, final String imageKey, final String key) {
011    super(key);
012    this.shape = shape;
013    this.imageKey = imageKey;
014  }
015
016  public String getImageKey() {
017    return imageKey;
018  }
019
020  public Path2D getShape() {
021    return shape;
022  }
023}
